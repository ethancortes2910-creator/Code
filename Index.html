import pygame import sys import random import os pygame.init() # ----------- Configurations ----------- # WIDTH, HEIGHT = 288, 512 screen = pygame.display.set_mode((WIDTH, HEIGHT)) clock = pygame.time.Clock() pygame.display.set_caption("Flappy Bird Ultimate Mobile Clone") # ----------- Constantes ----------- # GRAVITY = 0.25 BIRD_JUMP = -6 PIPE_SPEED = 2 PIPE_GAP = 100 BASE_SCROLL_SPEED = 2 # ----------- Assets ----------- # ASSETS = "assets" BG = pygame.image.load(os.path.join(ASSETS, "background.png")) BASE = pygame.image.load(os.path.join(ASSETS, "base.png")) BIRD_FRAMES = [ pygame.image.load(os.path.join(ASSETS, "bird1.png")), pygame.image.load(os.path.join(ASSETS, "bird2.png")), pygame.image.load(os.path.join(ASSETS, "bird3.png")), ] PIPE_IMG = pygame.image.load(os.path.join(ASSETS, "pipe.png")) jump_sound = pygame.mixer.Sound(os.path.join(ASSETS, "jump.wav")) point_sound = pygame.mixer.Sound(os.path.join(ASSETS, "point.wav")) hit_sound = pygame.mixer.Sound(os.path.join(ASSETS, "hit.wav")) # ----------- Variables ----------- # bird_index = 0 bird_x = 50 bird_y = HEIGHT//2 bird_movement = 0 pipe_list = [] score = 0 displayed_score = 0 best_score = 0 game_active = False base_x = 0 game_over_animation = False # ----------- Événements ----------- # BIRD_ANIM_EVENT = pygame.USEREVENT + 1 pygame.time.set_timer(BIRD_ANIM_EVENT, 200) SPAWNPIPE = pygame.USEREVENT + 2 pygame.time.set_timer(SPAWNPIPE, 1200) # ----------- Fonctions ----------- # def draw_pipes(pipes): for pipe in pipes: if pipe.bottom >= HEIGHT: screen.blit(PIPE_IMG, pipe) else: flip_pipe = pygame.transform.flip(PIPE_IMG, False, True) screen.blit(flip_pipe, pipe) def move_pipes(pipes): for pipe in pipes: pipe.centerx -= PIPE_SPEED return [pipe for pipe in pipes if pipe.right > -50] def check_collision(pipes): global game_active, game_over_animation, best_score bird_rect = BIRD_FRAMES[0].get_rect(center=(bird_x, bird_y)) for pipe in pipes: if bird_rect.colliderect(pipe): hit_sound.play() game_active = False game_over_animation = True if bird_y >= 450 or bird_y <= 0: hit_sound.play() game_active = False game_over_animation = True def rotate_bird(bird): return pygame.transform.rotozoom(bird, -bird_movement*3, 1) def show_menu(): font = pygame.font.SysFont(None, 30) start_text = font.render("Press SPACE to Start", True, (255,255,255)) screen.blit(start_text, (WIDTH//2 - start_text.get_width()//2, HEIGHT//2)) def score_display(current_score, best=False): font = pygame.font.SysFont(None, 40) text = f"Best: {current_score}" if best else str(int(current_score)) score_surface = font.render(text, True, (255,255,255)) y_pos = 450 if best else 50 screen.blit(score_surface, (WIDTH//2 - score_surface.get_width()//2, y_pos)) # ----------- Boucle principale ----------- # while True: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() if event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE: if game_active: bird_movement = BIRD_JUMP jump_sound.play() else: # Start / restart game_active = True pipe_list.clear() bird_y = HEIGHT//2 bird_movement = 0 score = 0 displayed_score = 0 game_over_animation = False if event.type == BIRD_ANIM_EVENT: bird_index = (bird_index + 1) % len(BIRD_FRAMES) if event.type == SPAWNPIPE and game_active: pipe_height = random.randint(150, 300) bottom_pipe = PIPE_IMG.get_rect(midtop=(WIDTH + 50, pipe_height)) top_pipe = PIPE_IMG.get_rect(midbottom=(WIDTH + 50, pipe_height - PIPE_GAP)) pipe_list.extend([bottom_pipe, top_pipe]) screen.blit(BG, (0,0)) if game_active: # Bird movement & rotation bird_movement += GRAVITY bird_y += bird_movement rotated_bird = rotate_bird(BIRD_FRAMES[bird_index]) bird_rect = rotated_bird.get_rect(center=(bird_x, bird_y)) screen.blit(rotated_bird, bird_rect) # Pipes pipe_list = move_pipes(pipe_list) draw_pipes(pipe_list) check_collision(pipe_list) # Score animation for pipe in pipe_list: if pipe.centerx == bird_x: score += 0.5 point_sound.play() if displayed_score < score: displayed_score += 0.05 score_display(displayed_score) else: # Game over bird animation if game_over_animation and bird_y < 450: bird_movement += GRAVITY bird_y += bird_movement rotated_bird = rotate_bird(BIRD_FRAMES[bird_index]) bird_rect = rotated_bird.get_rect(center=(bird_x, bird_y)) screen.blit(rotated_bird, bird_rect) else: game_over_animation = False if score > best_score: best_score = int(score) show_menu() score_display(best_score, best=True) # Base scrolling base_x -= BASE_SCROLL_SPEED if base_x <= -WIDTH: base_x = 0 screen.blit(BASE, (base_x, HEIGHT-100)) screen.blit(BASE, (base_x + WIDTH, HEIGHT-100)) pygame.display.update() clock.tick(120)
